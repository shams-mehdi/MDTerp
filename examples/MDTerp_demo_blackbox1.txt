def run_model(model, data): ## Do not change this line
    import numpy as np
    np.random.seed(0)
    relevant_indices = [0,1,3,4,7,8,9] ## MDTerp result should identify these feature indices as relevant with weights specified in the following line of code
    relevant_weights = [3,5,2,0.5,0.5,0.5,0.5]
    data = np.average(np.sin(data[:, relevant_indices]), weights = relevant_weights, axis = 1)
    predict = np.exp(-data)
    prediction = np.column_stack((predict, 1- predict ))
